const express = require('express');
const router = express.Router();
const { auth } = require('../middleware/auth');
const User = require('../models/User');
const Transaction = require('../models/Transaction');
const axios = require('axios');

const COINBASE_LEGACY_API = 'https://api.commerce.coinbase.com';
const COINBASE_BUSINESS_API = 'https://business.coinbase.com/api/v1';
const useBusinessAPI = process.env.USE_COINBASE_BUSINESS === 'true';

router.post('/create-charge', auth, async (req, res) => {
  try {
    const { amount, currency = 'USD' } = req.body;
    
    if (!amount || amount < 10) {
      return res.status(400).json({ message: 'Minimum deposit is $10' });
    }

    const apiKey = process.env.COINBASE_API_KEY;
    const businessApiToken = process.env.COINBASE_BUSINESS_TOKEN;
    
    if (!apiKey && !businessApiToken) {
      const mockChargeId = 'demo_' + Date.now();
      
      await Transaction.create({
        user: req.user.id,
        type: 'deposit',
        amount,
        currency,
        status: 'pending',
        coinbaseChargeId: mockChargeId,
        description: 'Demo deposit (API not configured)'
      });

      return res.json({
        chargeId: mockChargeId,
        code: 'DEMO_CODE',
        hostedUrl: `${process.env.FRONTEND_URL || 'http://localhost:3000'}/demo-payment?amount=${amount}`,
        expiresAt: new Date(Date.now() + 3600000).toISOString(),
        demo: true,
        message: 'Demo mode - Coinbase API not configured'
      });
    }

    let charge;
    
    if (useBusinessAPI && businessApiToken) {
      const response = await axios.post(
        `${COINBASE_BUSINESS_API}/checkouts`,
        {
          name: 'SureReturns Deposit',
          description: `Deposit $${amount} to your SureReturns account`,
          pricing_type: 'fixed_price',
          local_price: {
            amount: amount.toString(),
            currency: currency
          },
          requested_info: ['email'],
          metadata: {
            user_id: req.user.id.toString(),
            email: req.user.email
          },
          redirect_url: `${process.env.FRONTEND_URL || 'http://localhost:3000'}/dashboard`,
          cancel_url: `${process.env.FRONTEND_URL || 'http://localhost:3000'}/deposit`
        },
        {
          headers: {
            'Authorization': `Bearer ${businessApiToken}`,
            'Content-Type': 'application/json',
            'X-CC-Version': '2018-03-22'
          }
        }
      );
      
      charge = response.data.data || response.data;
    } else {
      const response = await axios.post(
        `${COINBASE_LEGACY_API}/charges`,
        {
          name: 'SureReturns Deposit',
          description: `Deposit $${amount} to your SureReturns account`,
          local_price: {
            amount: amount.toString(),
            currency: currency
          },
          pricing_type: 'fixed_price',
          metadata: {
            user_id: req.user.id.toString(),
            email: req.user.email
          },
          redirect_url: `${process.env.FRONTEND_URL || 'http://localhost:3000'}/dashboard`,
          cancel_url: `${process.env.FRONTEND_URL || 'http://localhost:3000'}/deposit`
        },
        {
          headers: {
            'X-CC-Api-Key': apiKey,
            'X-CC-Version': '2018-03-22',
            'Content-Type': 'application/json'
          }
        }
      );

      charge = response.data.data;
    }

    await Transaction.create({
      user: req.user.id,
      type: 'deposit',
      amount,
      currency,
      status: 'pending',
      coinbaseChargeId: charge.id,
      coinbaseChargeCode: charge.code,
      description: `Deposit via ${useBusinessAPI ? 'Coinbase Business' : 'Coinbase Commerce'}`
    });

    await User.findByIdAndUpdate(req.user.id, {
      $push: {
        depositHistory: {
          amount,
          currency,
          status: 'pending',
          coinbaseChargeId: charge.id
        }
      }
    });

    res.json({
      chargeId: charge.id,
      code: charge.code,
      hostedUrl: charge.hosted_url,
      expiresAt: charge.expires_at,
      api: useBusinessAPI ? 'business' : 'commerce'
    });
  } catch (error) {
    console.error('Coinbase Error:', error.response?.data || error.message);
    
    if (error.response?.status === 401) {
      return res.status(500).json({ 
        message: 'Invalid API credentials. Please check your Coinbase API key.' 
      });
    }
    
    res.status(500).json({ 
      message: error.response?.data?.error?.message || 'Failed to create payment charge' 
    });
  }
});

router.post('/webhook', async (req, res) => {
  try {
    const event = req.body;
    
    console.log('Webhook received:', event.event?.type || event.type);

    const eventType = event.event?.type || event.type;
    
    if (eventType === 'charge:confirmed' || eventType === 'checkout:completed') {
      const charge = event.event?.data || event.data;
      const metadata = charge.metadata || {};
      const userId = metadata.user_id;
      const amount = parseFloat(charge.pricing?.local?.amount || charge.amount);
      const chargeId = charge.id;

      if (userId && amount > 0) {
        const user = await User.findById(userId);
        if (user) {
          user.balance += amount;
          await user.save();

          await Transaction.findOneAndUpdate(
            { coinbaseChargeId: chargeId },
            { 
              status: 'completed',
              description: 'Deposit confirmed via webhook'
            }
          );

          await User.findByIdAndUpdate(userId, {
            $set: {
              'depositHistory.$[elem].status': 'completed'
            }
          }, {
            arrayFilters: [{ 'elem.coinbaseChargeId': chargeId }]
          });

          console.log(`Credited $${amount} to user ${userId}`);
        }
      }
    }

    if (eventType === 'charge:failed' || eventType === 'checkout:failed') {
      const charge = event.event?.data || event.data;
      await Transaction.findOneAndUpdate(
        { coinbaseChargeId: charge.id },
        { status: 'failed', description: 'Payment failed' }
      );
    }

    res.status(200).json({ received: true });
  } catch (error) {
    console.error('Webhook Error:', error);
    res.status(200).json({ received: true, error: error.message });
  }
});

router.get('/status/:chargeId', auth, async (req, res) => {
  try {
    const transaction = await Transaction.findOne({
      coinbaseChargeId: req.params.chargeId,
      user: req.user.id
    });

    if (!transaction) {
      return res.status(404).json({ message: 'Transaction not found' });
    }

    res.json({
      status: transaction.status,
      amount: transaction.amount,
      createdAt: transaction.createdAt,
      type: transaction.type
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ message: 'Server error' });
  }
});

module.exports = router;
