import React, { useState } from 'react';
import { motion } from 'framer-motion';
import { 
  DollarSign, 
  ArrowRight, 
  AlertCircle, 
  CheckCircle2,
  Copy,
  ExternalLink,
  Loader2
} from 'lucide-react';
import { useAuth } from '../context/AuthContext';
import axios from 'axios';
import { toast } from 'react-toastify';

const Deposit = () => {
  const [amount, setAmount] = useState('');
  const [loading, setLoading] = useState(false);
  const [charge, setCharge] = useState(null);
  const { user } = useAuth();

  const handleCreateCharge = async (e) => {
    e.preventDefault();
    
    if (!amount || amount < 10) {
      toast.error('Minimum deposit is $10');
      return;
    }

    setLoading(true);
    try {
      const res = await axios.post('/api/payment/create-charge', {
        amount: parseFloat(amount),
        currency: 'USD'
      });
      
      setCharge(res.data);
      toast.success('Payment link created! Redirecting to Coinbase...');
      
      if (res.data.hostedUrl) {
        window.open(res.data.hostedUrl, '_blank');
      }
    } catch (error) {
      console.error('Payment error:', error);
      toast.error(error.response?.data?.message || 'Failed to create payment');
    } finally {
      setLoading(false);
    }
  };

  const copyToClipboard = (text) => {
    navigator.clipboard.writeText(text);
    toast.success('Copied to clipboard!');
  };

  return (
    <div className="min-h-screen pt-20 pb-12 px-4 sm:px-6 lg:px-8">
      <div className="max-w-2xl mx-auto">
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          className="glass-card p-8"
        >
          <div className="text-center mb-8">
            <div className="w-16 h-16 bg-blue-500/20 rounded-full flex items-center justify-center mx-auto mb-4">
              <DollarSign className="w-8 h-8 text-blue-400" />
            </div>
            <h1 className="text-3xl font-bold mb-2">Deposit Funds</h1>
            <p className="text-gray-400">
              Add funds to your account using cryptocurrency
            </p>
          </div>

          {!charge ? (
            <form onSubmit={handleCreateCharge} className="space-y-6">
              <div>
                <label className="block text-sm font-medium text-gray-300 mb-2">
                  Deposit Amount (USD)
                </label>
                <div className="relative">
                  <DollarSign className="absolute left-3 top-1/2 transform -translate-y-1/2 w-5 h-5 text-gray-400" />
                  <input
                    type="number"
                    min="10"
                    step="0.01"
                    value={amount}
                    onChange={(e) => setAmount(e.target.value)}
                    className="input-field pl-10"
                    placeholder="Enter amount (min $10)"
                    required
                  />
                </div>
                <p className="text-sm text-gray-500 mt-2">
                  Minimum deposit: $10. No maximum limit.
                </p>
              </div>

              <div className="p-4 bg-blue-500/10 border border-blue-500/20 rounded-lg">
                <div className="flex items-start space-x-3">
                  <AlertCircle className="w-5 h-5 text-blue-400 mt-0.5" />
                  <div className="text-sm text-gray-300">
                    <p className="font-medium text-white mb-1">How it works:</p>
                    <ul className="space-y-1 list-disc list-inside">
                      <li>Enter your deposit amount</li>
                      <li>You'll be redirected to Coinbase secure payment page</li>
                      <li>Pay with BTC, ETH, USDC, or other supported crypto</li>
                      <li>Funds are credited automatically after confirmation</li>
                    </ul>
                  </div>
                </div>
              </div>

              <button
                type="submit"
                disabled={loading}
                className="w-full btn-primary py-4 flex items-center justify-center space-x-2 disabled:opacity-50"
              >
                {loading ? (
                  <>
                    <Loader2 className="w-5 h-5 animate-spin" />
                    <span>Creating Payment...</span>
                  </>
                ) : (
                  <>
                    <span>Continue to Payment</span>
                    <ArrowRight className="w-5 h-5" />
                  </>
                )}
              </button>
            </form>
          ) : (
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              className="space-y-6"
            >
              <div className="p-4 bg-green-500/10 border border-green-500/20 rounded-lg">
                <div className="flex items-center space-x-3">
                  <CheckCircle2 className="w-6 h-6 text-green-400" />
                  <div>
                    <p className="font-medium text-white">Payment Link Created!</p>
                    <p className="text-sm text-gray-400">Complete your payment on Coinbase</p>
                  </div>
                </div>
              </div>

              <div className="space-y-4">
                <div className="p-4 bg-gray-800/50 rounded-lg">
                  <label className="text-sm text-gray-400 mb-2 block">Payment URL</label>
                  <div className="flex items-center space-x-2">
                    <input
                      type="text"
                      value={charge.hostedUrl}
                      readOnly
                      className="input-field flex-1 text-sm"
                    />
                    <button
                      onClick={() => copyToClipboard(charge.hostedUrl)}
                      className="p-2 hover:bg-gray-700 rounded-lg transition-colors"
                    >
                      <Copy className="w-5 h-5 text-gray-400" />
                    </button>
                  </div>
                </div>
              </div>

              <a
                href={charge.hostedUrl}
                target="_blank"
                rel="noopener noreferrer"
                className="w-full btn-primary py-4 flex items-center justify-center space-x-2"
              >
                <span>Open Payment Page</span>
                <ExternalLink className="w-5 h-5" />
              </a>

              <button
                onClick={() => {
                  setCharge(null);
                  setAmount('');
                }}
                className="w-full py-3 border border-gray-600 rounded-lg text-gray-400 hover:text-white hover:border-gray-500 transition-all"
              >
                Create New Payment
              </button>
            </motion.div>
          )}
        </motion.div>
      </div>
    </div>
  );
};

export default Deposit;
